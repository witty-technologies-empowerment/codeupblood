[Unit]
Description=gunicorn daemon
After=network.target

[Service]
User=cub
Group=www-data
WorkingDirectory=/home/cub/codeupblood
ExecStart=/home/cub/codeupblood/venv/bin/gunicorn --access-logfile - --workers 3 --bind unix:/home/cub/codeupblood/cub.sock src.wsgi:application

[Install]
WantedBy=multi-user.target







server {
    listen 80;
    server_name 172.104.8.16 codeupblood.com;

    location = /favicon.ico { access_log off; log_not_found off; }
    location /static/ {
        root /home/cub/codeupblood;
    }

    location / {
        include proxy_params;
        proxy_pass http://unix:/home/cub/codeupblood/cub.sock;
    }
}

server_name codeupblood.com www.codeupblood.com;

sudo certbot --nginx -d codeupblood.com -d www.codeupblood.com
